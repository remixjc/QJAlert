全景视觉 alert 实例

1.弹窗

代码：

            
                    qjalert('文本');//替代alert();

2.弹窗(带标题)

代码：

            
                    qjalert('这是标题！','这是文本');

3.弹窗(成功的提示)

代码：

            
                    qjalert('这还是标题！','这还是文本','success');

4.弹窗(带函数传值的alert)

代码：

            
                    qjalert({ 
                        title: '确定删除吗？', 
                        text: '你将无法恢复该虚拟文件！', 
                        type: 'warning',
                        showCancelButton: true, 
                        confirmButtonColor: '#DD6B55',
                        confirmButtonText: '确定删除！', 
                        cancelButtonText: '取消删除！',
                        closeOnConfirm: false, 
                        closeOnCancel: false	
                        },
                        function(isConfirm){ 
                        if (isConfirm) { 
                            qjalert('删除！', '你的虚拟文件已经被删除。',
                        'success'); 
                        } else { 
                            qjalert('取消！', '你的虚拟文件是安全的:)',
                        'error'); 
                        } 
                        });

5.弹窗(自定义文本)

代码：

            
                    qjalert({ 
                        title: 'HTML 标题!',
                        text: '自定义这是蓝色的。', 
                        html: true 
                    });
